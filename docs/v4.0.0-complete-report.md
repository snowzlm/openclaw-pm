  .command('health')
  .description('执行 Gateway 健康检查')
  .option('-j, --json', '输出 JSON 格式')
  .action(async (options) => {
    const checker = new GatewayHealthChecker(config);

  });

program

  .description('创建备份')
  .option('-t, --type <type>', '备份类型 (manual|auto)', 'manual')
