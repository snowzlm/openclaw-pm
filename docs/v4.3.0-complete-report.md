```typescript
export class StatsGenerator {
  private config: ConfigManager;
  private logger: Logger;
  private healthChecker: GatewayHealthChecker;

  async generateDailyStats(date: string): Promise<DailyStats> {
    // 1. 读取日志文件
    const logPath = this.getLogPath(date);
    const logContent = fs.readFileSync(logPath, 'utf-8');

    // 2. 分析日志内容
    const messages = this.analyzeMessages(logContent);
    const hourlyDistribution = this.analyzeHourlyDistribution(logContent);
    const errors = this.analyzeErrors(logContent);
    const gateway = this.analyzeGateway(logContent);

    // 3. 计算健康评分
    const health = this.calculateHealthScore({
      messages,
      errors,
      gateway
    });

    return { date, messages, hourlyDistribution, errors, gateway, health };
  }
}
```

### 2. 健康评分算法
```typescript
private calculateHealthScore(stats: any): any {
  let score = 100;
  
  // 错误扣分
  if (stats.errors.total > 0) {
    score -= Math.min(stats.errors.total * 2, 30);
  }
  
  // Gateway 重启扣分
  if (stats.gateway.starts > 1) {
    score -= (stats.gateway.starts - 1) * 10;
  }
  
  // 活跃度评估
  const activity = stats.messages.received > 100 ? '高' : 
                   stats.messages.received > 50 ? '中' : '低';
  
  // 稳定性评估
  const stability = stats.gateway.starts <= 1 && stats.errors.total === 0 ? 
                    '优秀' : '一般';
  
  return { score: Math.max(score, 0), activity, stability };
}
```
